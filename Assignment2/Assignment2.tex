\section{Problem}
Suppose $A$ and $B$ are two independent events with probabilities $\Pr(A) \neq 0$ and $\Pr(B) \neq 0$.
Let $\complement{A}$ and $\complement{B}$  be their complements. Which one of the following statements is FALSE?
\begin{enumerate}
    \item $\Pr(A \cap B)$ = $\Pr(A)\Pr(B)$
    \item $\Pr(A|B) = \Pr(A)$
    \item $\Pr(A \cup B) = \Pr(A) + \Pr(B)$
    \item $\Pr(\complement{A} \cap \complement{B}) = \Pr(\complement{A})\Pr(\complement{B})$
\end{enumerate}
\section{Solution}
\begin{enumerate}
    \item As $A,B$ are independent events, By definition,  
    \begin{align*}
        \Pr(A \cap B) &= \Pr(A)\Pr(B)
    \end{align*}
    Thus option 1 is true.
    \item 
    \begin{align*}
        \Pr(A|B) &= \frac{\Pr(A \cap B)}{\Pr(B)} \\
        &= \frac{\Pr(A)\Pr(B)}{\Pr(B)}\\
        &= \Pr(A)
    \end{align*}
    Thus option 2 is true.
    \item \begin{align*}
        \Pr(A \cup B) &= \Pr(A) + \Pr(B) - \Pr(A \cap B) \\
        &= \Pr(A) + \Pr(B) - \Pr(A)\Pr(B)
    \end{align*}
    Thus option 3 is false. 
    \item \begin{align*}
        \Pr(\complement{A} \cap \complement{B}) &= \Pr(\complement{A \cup B} )\\
        &= 1- \Pr( A \cup B)\\
        &= 1- \Pr(A) - \Pr(B) + \Pr(A \cap B) \\
        &= ( 1- \Pr(A) ) ( 1- \Pr(B) ) \\
        &= \Pr(\complement{A})\Pr(\complement{B})
    \end{align*}
    Thus option 4 is true.
\end{enumerate}
Hence, FALSE statement is option 3.
